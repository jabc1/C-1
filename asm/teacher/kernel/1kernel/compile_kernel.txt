. 生成一个默认配置
	cp /boot/config-`uname -r` .config

	make defconfig

	cp arch/x86/configs/
	cp arch/arm/configs/

. 再配置
	make menuconfig

. 编译
	make dep
	make clean
	make bzImage	-> 只编译-y, 编译进内核
	make modules	-> 只编译模块

	or:

	make

	(.o .ko vmlinux arch/arm/boot/bzImage)

. 安装
	make modules_install	(.ko -> /lib/modules/linux-2.6.36/)
	make install	(bzImage -> /boot, grub)
